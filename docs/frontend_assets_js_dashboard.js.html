/**
 * Récupère le token stocké localement.
 * Si absent, redirige vers la page de login.
 */
const token = localStorage.getItem("token");
if (!token) {
  window.location.href = "index.html";
}

/**
 * Interroge l’API pour récupérer le rôle de l’utilisateur connecté.
 *
 * @async
 * @returns {Promise<string>} Le rôle utilisateur ("admin" ou "user").
 * @throws {Error} Si la requête échoue ou si non autorisé.
 */
async function fetchUserRole() {
  try {
    const res = await fetch("http://localhost:8080/api/auth/me", {
      headers: { Authorization: "Bearer " + token },
    });
    if (!res.ok) throw new Error("Non autorisé");
    const data = await res.json();
    return data.role; // ex: "admin" ou "user"
  } catch (error) {
    alert("Erreur d'authentification. Veuillez vous reconnecter.");
    logout();
  }
}

/**
 * Supprime le token et redirige vers la page de login.
 */
function logout() {
  localStorage.removeItem("token");
  window.location.href = "index.html";
}
