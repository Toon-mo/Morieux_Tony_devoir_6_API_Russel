/**
 * Middleware pour vérifier que l'utilisateur a le rôle "admin".
 *
 * @function checkAdmin
 * @param {Object} req - Objet requête Express, doit contenir `user` avec propriété `role`.
 * @param {Object} res - Objet réponse Express.
 * @param {Function} next - Fonction pour passer au middleware suivant.
 *
 * @returns {void}
 * @throws Renvoie un statut 403 avec message si l'utilisateur n'est pas admin.
 */
module.exports = (req, res, next) => {
  if (req.user?.role === "admin") {
    next();
  } else {
    res
      .status(403)
      .json({ message: "Accès interdit : rôle administrateur requis" });
  }
};
